To build the user table - Used by DB team to build test DB

CREATE TABLE USERS (
Username VARCHAR(45) PRIMARY KEY NOT NULL,
Name VARCHAR(45) NOT NULL,
Password VARCHAR(45) NOT NULL,
Project_Name VARCHAR(45) NOT NULL
);

To build the workGroup table - Used by DB team to build test DB

CREATE TABLE WORKGROUPS (
ID INT PRIMARY KEY NOT NULL,
Name VARCHAR(45) NOT NULL
);

To build the userWorkGroup table - Used by DB team to build test DB

CREATE TABLE USER_WORKGROUP (
Username VARCHAR(45) NOT NULL,
WGID INT NOT NULL,
FOREIGN KEY fk(Username) REFERENCES USERS(Username),
FOREIGN KEY fkk(WGID) REFERENCES WORKGROUPS(ID),
CONSTRAINT PK_group PRIMARY KEY (Username,WGID)
);

To build the cases table - Used by DB team to build test DB

CREATE TABLE CASES (
Case_Number INT PRIMARY KEY NOT NULL,
Street VARCHAR(45) NOT NULL,
City VARCHAR(45) NOT NULL,
State VARCHAR(45) NOT NULL,
Zip VARCHAR(10) NOT NULL,
Date_Of_Accident VARCHAR(45) NOT NULL,
Time VARCHAR(45) NOT NULL,
Lead_Investigator VARCHAR(45) NOT NULL,
WGID INT NOT NULL,
Creator_ID VARCHAR(45) NOT NULL,
FOREIGN KEY fk(Lead_Investigator) REFERENCES USERS(Username),
FOREIGN KEY fkk(WGID) REFERENCES WORKGROUPS(ID),
FOREIGN KEY fkkk(Creator_ID) REFERENCES USERS(Username)
);

To build the vehicle table - Used by DB team to build test DB

CREATE TABLE VEHICLE (
VIN VARCHAR(45) PRIMARY KEY NOT NULL,
Case_Number INT NOT NULL,
License_Plate_Number VARCHAR(10) NOT NULL,
Make VARCHAR(45) NOT NULL,
Model VARCHAR(45) NOT NULL,
Year VARCHAR(4) NOT NULL,
Speed INT NOT NULL,
Airbag_Deploy VARCHAR(1) NOT NULL,
Color VARCHAR(45) NOT NULL,
License_Plate_Expiration VARCHAR(45) NOT NULL,
License_Plate_State VARCHAR(15) NOT NULL,
FOREIGN KEY fk(Case_Number) REFERENCES CASES(Case_Number)
);

To build the occupant table - Used by DB team to build test DB

CREATE TABLE OCCUPANT_INFO (
ID INT PRIMARY KEY NOT NULL,
Name VARCHAR(45) UNIQUE NOT NULL,
Impairment VARCHAR(45) NOT NULL,
Seatbelt VARCHAR(1) NOT NULL,
Phone_No VARCHAR(10) NOT NULL,
DOB VARCHAR(45) NOT NULL,
Driving VARCHAR(45) NOT NULL,
FOREIGN KEY fk(Driving) REFERENCES VEHICLE(VIN)
);

To build the reference table - Used by DB team to build test DB

CREATE TABLE REFERENCE (
Case_ID INT PRIMARY KEY NOT NULL,
Witness_Notes VARCHAR(1000) NOT NULL,
Personal_Notes VARCHAR(1000) NOT NULL,
FOREIGN KEY fk(Case_ID) REFERENCES CASES(Case_Number)
);

To build the environment table - Used by DB team to build test DB

CREATE TABLE ENVIRONMENT (
Case_Number INT PRIMARY KEY NOT NULL,
Precipitation VARCHAR(100) NOT NULL,
Lighting VARCHAR(100) NOT NULL,
Temperature VARCHAR(100) NOT NULL,
Weather VARCHAR(100) NOT NULL,
Roadway_Conditions VARCHAR(100) NOT NULL,
FOREIGN KEY fk(Case_Number) REFERENCES CASES(Case_Number)
);
